// GÜNCEL VE STABİL SAYI TEMİZLEME FONKSİYONU function parseToFloat(v) { if (v === undefined || v === null || v === "" || v === 0) return 0; if (typeof v === 'number') return v; let s = String(v).trim(); // Eğer sayı hem nokta hem virgül içeriyorsa (Örn: 1.250,50) if (s.includes(',') && s.includes('.')) { // Noktaları (binlik ayırıcı) sil, virgülü noktaya çevir s = s.replace(/\./g, '').replace(',', '.'); } // Sadece virgül varsa (Örn: 1250,50) else if (s.includes(',')) { s = s.replace(',', '.'); } // Rakamlar, nokta ve eksi dışındaki her şeyi (₺, boşluk vb.) temizle s = s.replace(/[^0-9.-]/g, ''); let num = parseFloat(s); return isNaN(num) ? 0 : num; } // GÜNCEL PARA FORMATLAMA FONKSİYONU function formatTRY(v) { let num = parseToFloat(v); return new Intl.NumberFormat('tr-TR', { style: 'currency', currency: 'TRY', minimumFractionDigits: 2 }).format(num); }
